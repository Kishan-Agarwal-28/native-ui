## Usage

```tsx
import { Textarea } from '@/components/ui/textarea';
```

```tsx
<Textarea
  placeholder="Enter your message…"
  value={message}
  onChangeText={setMessage}
/>
```

## Examples

### Controlled

```tsx
const [bio, setBio] = useState('');

<Textarea
  value={bio}
  onChangeText={setBio}
  placeholder="Tell us about yourself"
  style={{ minHeight: 100 }}
/>
```

### Error state

```tsx
<Textarea
  placeholder="Required"
  aria-invalid={!!errors.message}
  value={message}
  onChangeText={setMessage}
/>
{errors.message && (
  <Text style={{ color: 'red', fontSize: 12 }}>{errors.message}</Text>
)}
```

### Read-only

```tsx
<Textarea
  value="This content cannot be changed."
  editable={false}
/>
```

### Inside a form field

```tsx
<Field error={errors.notes}>
  <FieldLabel>Notes</FieldLabel>
  <Textarea
    placeholder="Add any additional notes…"
    value={notes}
    onChangeText={setNotes}
    style={{ minHeight: 80 }}
    aria-invalid={!!errors.notes}
  />
  <FieldError />
</Field>
```

### Fixed height

```tsx
<Textarea
  placeholder="Short description"
  style={{ minHeight: 48, maxHeight: 48 }}
  numberOfLines={2}
/>
```

## API Reference

### Textarea

Extends all React Native `TextInputProps`.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `aria-invalid` | `boolean` | — | When `true`, renders the border in `colors.destructive`. |
| `editable` | `boolean` | `true` | When `false`, dims the field and applies a subtle disabled background. |
| `placeholder` | `string` | — | Placeholder text, styled with `colors.mutedForeground`. |
| `style` | `TextStyle` | — | Additional styles. Use `minHeight` to control the default size (defaults to `64px`). |
| `onFocus` | `(e) => void` | — | Called on focus; also switches border to `colors.ring`. |
| `onBlur` | `(e) => void` | — | Called on blur; border reverts to `colors.border`. |

> `multiline` and `textAlignVertical="top"` are set internally — you do not need to pass them.
