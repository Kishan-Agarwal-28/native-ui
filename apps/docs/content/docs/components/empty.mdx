## Usage

```tsx
import {
  Empty,
  EmptyHeader,
  EmptyMedia,
  EmptyTitle,
  EmptyDescription,
  EmptyContent,
} from '@/components/ui/empty';
```

```tsx
<Empty>
  <EmptyHeader>
    <EmptyTitle>No items yet</EmptyTitle>
    <EmptyDescription>Add your first item to get started.</EmptyDescription>
  </EmptyHeader>
</Empty>
```

## Examples

### Icon variant

```tsx
<Empty>
  <EmptyHeader>
    <EmptyMedia variant="icon">
      <Feather name="file-text" size={28} color="#71717a" />
    </EmptyMedia>
    <EmptyTitle>No documents</EmptyTitle>
    <EmptyDescription>Upload a document to see it here.</EmptyDescription>
  </EmptyHeader>
  <EmptyContent>
    <Button title="Upload" />
  </EmptyContent>
</Empty>
```

### Image variant

```tsx
<Empty>
  <EmptyHeader>
    <EmptyMedia variant="image">
      <Image source={require('@/assets/empty.png')} style={{ width: 120, height: 120 }} />
    </EmptyMedia>
    <EmptyTitle>Nothing here</EmptyTitle>
    <EmptyDescription>Come back later.</EmptyDescription>
  </EmptyHeader>
</Empty>
```

### Minimal (title + description only)

```tsx
<Empty>
  <EmptyHeader>
    <EmptyTitle>No notifications</EmptyTitle>
    <EmptyDescription>You're all caught up!</EmptyDescription>
  </EmptyHeader>
</Empty>
```

## API Reference

### Empty

Root container. Centered, full-flex layout with padding.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `style` | `ViewStyle` | — | Additional styles. |

### EmptyHeader

Centered column for the media, title, and description.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `style` | `ViewStyle` | — | Additional styles. |

### EmptyMedia

Container for an illustration or icon. The `"icon"` variant renders a rounded background pill; `"image"` is a plain wrapper.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `variant` | `"icon" \| "image"` | `"icon"` | Visual container style. |
| `style` | `ViewStyle` | — | Additional styles. |

### EmptyTitle

Large, centered heading text.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `style` | `TextStyle` | — | Additional text styles. |

### EmptyDescription

Smaller, centered muted description text.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `style` | `TextStyle` | — | Additional text styles. |

### EmptyContent

Centered action area (e.g., buttons).

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `style` | `ViewStyle` | — | Additional styles. |
