## Usage

```tsx
import { ButtonGroup, ButtonGroupText, ButtonGroupSeparator } from '@/components/ui/button-group';
import Button from '@/components/ui/button';
```

```tsx
<ButtonGroup>
  <Button title="Left" variant="outline" />
  <Button title="Right" variant="outline" />
</ButtonGroup>
```

## Examples

### Horizontal group (default)

```tsx
<ButtonGroup orientation="horizontal">
  <Button title="Bold" variant="outline" />
  <Button title="Italic" variant="outline" />
  <Button title="Underline" variant="outline" />
</ButtonGroup>
```

### Vertical group

```tsx
<ButtonGroup orientation="vertical">
  <Button title="Option A" variant="outline" />
  <Button title="Option B" variant="outline" />
  <Button title="Option C" variant="outline" />
</ButtonGroup>
```

### Mixed with label

Use `ButtonGroupText` to add a non-interactive label inside the group:

```tsx
<ButtonGroup>
  <ButtonGroupText>Sort by:</ButtonGroupText>
  <Button title="Name" variant="outline" />
  <Button title="Date" variant="outline" />
</ButtonGroup>
```

### With separator

```tsx
<ButtonGroup>
  <Button title="Cut" variant="outline" />
  <ButtonGroupSeparator />
  <Button title="Copy" variant="outline" />
  <ButtonGroupSeparator />
  <Button title="Paste" variant="outline" />
</ButtonGroup>
```

## API Reference

### ButtonGroup

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `orientation` | `"horizontal" \| "vertical"` | `"horizontal"` | Direction of the button group. |
| `style` | `ViewStyle` | — | Additional container styles. |
| `children` | `ReactNode` | — | Button elements to group. |

### ButtonGroupText

A non-interactive label element styled to match the group height.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `icon` | `ReactNode` | — | Optional leading icon. |
| `style` | `ViewStyle` | — | Additional container styles. |
| `textStyle` | `TextStyle` | — | Additional text styles. |
| `children` | `string \| ReactNode` | — | Label content. |

### ButtonGroupSeparator

A thin divider line between group items. Automatically orients based on the group direction.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `orientation` | `"horizontal" \| "vertical"` | Inherits from group | Override the separator orientation. |
| `style` | `ViewStyle` | — | Additional styles. |
