## Usage

```tsx
import {
  FieldSet,
  FieldLegend,
  FieldGroup,
  Field,
  FieldLabel,
  FieldDescription,
  FieldError,
  FieldSeparator,
} from '@/components/ui/field';
```

```tsx
<Field error={errors.email}>
  <FieldLabel>Email</FieldLabel>
  <Input placeholder="you@example.com" inputMode="email" />
  <FieldError />
</Field>
```

## Examples

### Horizontal layout

```tsx
<Field orientation="horizontal">
  <FieldLabel>Dark Mode</FieldLabel>
  <Switch value={darkMode} onValueChange={setDarkMode} />
</Field>
```

### Grouped fields with a legend

```tsx
<FieldSet>
  <FieldLegend>Billing Address</FieldLegend>
  <FieldGroup>
    <Field>
      <FieldLabel>Street</FieldLabel>
      <Input placeholder="123 Main St" />
    </Field>
    <Field>
      <FieldLabel>City</FieldLabel>
      <Input placeholder="New York" />
    </Field>
  </FieldGroup>
</FieldSet>
```

### With separator

```tsx
<FieldGroup>
  <Field>
    <FieldLabel>First Name</FieldLabel>
    <Input placeholder="Jane" />
  </Field>
  <FieldSeparator />
  <Field>
    <FieldLabel>Last Name</FieldLabel>
    <Input placeholder="Smith" />
  </Field>
</FieldGroup>
```

### Error state

`FieldError` reads the `error` prop from the parent `Field` context and renders only when an error is present.

```tsx
<Field error={formErrors.username}>
  <FieldLabel>Username</FieldLabel>
  <Input value={username} onChangeText={setUsername} />
  <FieldError />
  {/* renders the error message, or nothing if error is undefined */}
</Field>
```

## API Reference

### FieldSet

A semantic grouping container. Adds `gap: 16` between children.

### FieldLegend

A bold heading for a `FieldSet`.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `style` | `TextStyle` | — | Additional text styles. |

### FieldGroup

Groups multiple `Field` components with `gap: 24`.

### Field

The main wrapper that provides context to `FieldError`.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `orientation` | `"vertical" \| "horizontal"` | `"vertical"` | Layout direction for label + control. |
| `error` | `string` | — | Error message. Passed to `FieldError` via context. |
| `style` | `ViewStyle` | — | Additional container styles. |

### FieldLabel

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `style` | `TextStyle` | — | Additional text styles. |

### FieldDescription

Muted helper text rendered below the control.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `style` | `TextStyle` | — | Additional text styles. |

### FieldError

Reads `error` from the nearest `Field` context. Renders nothing when no error is present.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `style` | `TextStyle` | — | Additional text styles. |

### FieldSeparator

A 1px horizontal divider using `colors.border`.

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `style` | `ViewStyle` | — | Additional styles. |
